إعداد Twilio — استقبال المكالمات

احصل على رقم هاتف يستقبل المكالمات ويوجهها للنظام
1
إنشاء حساب Twilio

روح على twilio.com وسجّل حساب مجاني — تحصل على $15 رصيد مجاني للتجربة

2
شراء رقم هاتف

من Console → Phone Numbers → Buy a Number. اختر رقم يدعم Voice. التكلفة ~$1/شهر

3
احفظ بيانات الـ API

من Dashboard احفظ: Account SID و Auth Token

4
إعداد Webhook

في إعدادات الرقم → Voice → Webhook URL: ضع رابط السيرفر الخاص بك (ستضيفه في الخطوة 6)

مثال: https://your-app.railway.app/call

كود السيرفر الكامل (Python)

ملف واحد يربط كل الخدمات معاً
📦

تحتاج Python 3.9+ مثبت. ثبّت المكتبات بـ: pip install flask twilio openai elevenlabs requests

PYTHON — app.py 
# ══════════════════════════════════════════════════
# SMART VOICE RECEPTION AI — السيرفر الرئيسي
# ══════════════════════════════════════════════════

import os, tempfile, requests
from flask import Flask, request, Response
from twilio.twiml.voice_response import VoiceResponse, Gather
from openai import OpenAI

app = Flask(__name__)
client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

# ── إعدادات المطعم ──────────────────────────────────
RESTAURANT_PROMPT = """
أنت مساعد صوتي ذكي لمطعم "الزيتون الذهبي".
- تتحدث بالعربية فقط
- أسلوبك ودود ومهذب
- تستقبل طلبات التوصيل والاستفسارات
- المنيو: شاورما 14₪، فلافل 8₪، مندي 35₪، برغر 20₪
- التوصيل مجاني للطلبات فوق 30₪، وإلا 5₪
- وقت التوصيل 45-60 دقيقة
- إذا بدا الزبون مستاءً، أعطه خصم 10% تلقائياً
- اسأل دائماً: الطلب، العنوان، رقم الهاتف
"""

conversation_history = {}

# ── استقبال المكالمة ────────────────────────────────
@app.route("/call", methods=["POST"])
def handle_call():
    call_sid = request.form.get("CallSid")
    conversation_history[call_sid] = []
    
    resp = VoiceResponse()
    gather = Gather(
        input="speech",
        action="/process",
        language="ar-SA",
        speech_timeout="auto",
        enhanced="true"
    )
    # ترحيب أولي
    audio_url = text_to_speech("أهلاً وسهلاً، مطعم الزيتون الذهبي. كيف أقدر أساعدك؟")
    gather.play(audio_url)
    resp.append(gather)
    return Response(str(resp), mimetype="text/xml")

# ── معالجة كلام الزبون ──────────────────────────────
@app.route("/process", methods=["POST"])
def process_speech():
    call_sid = request.form.get("CallSid")
    speech_text = request.form.get("SpeechResult", "")
    
    if not speech_text:
        return repeat_gather("لم أفهم، ممكن تعيد؟")
    
    # أضف كلام الزبون للتاريخ
    history = conversation_history.get(call_sid, [])
    history.append({"role": "user", "content": speech_text})
    
    # احصل على رد GPT-4
    ai_response = get_ai_response(history)
    history.append({"role": "assistant", "content": ai_response})
    conversation_history[call_sid] = history
    
    # حوّل الرد لصوت وشغّله
    audio_url = text_to_speech(ai_response)
    resp = VoiceResponse()
    
    # إذا انتهى الطلب
    if "شكراً" in ai_response or "مع السلامة" in ai_response:
        resp.play(audio_url)
        resp.hangup()
    else:
        gather = Gather(input="speech", action="/process",
                         language="ar-SA", speech_timeout="auto")
        gather.play(audio_url)
        resp.append(gather)
    
    return Response(str(resp), mimetype="text/xml")

# ── GPT-4 ────────────────────────────────────────────
def get_ai_response(history):
    messages = [{"role": "system", "content": RESTAURANT_PROMPT}] + history
    response = client.chat.completions.create(
        model="gpt-4o",
        messages=messages,
        max_tokens=200,
        temperature=0.7
    )
    return response.choices[0].message.content

# ── ElevenLabs TTS ──────────────────────────────────
def text_to_speech(text):
    VOICE_ID = os.environ.get("ELEVENLABS_VOICE_ID", "YOUR_VOICE_ID")
    API_KEY  = os.environ["ELEVENLABS_API_KEY"]
    
    response = requests.post(
        f"https://api.elevenlabs.io/v1/text-to-speech/{VOICE_ID}",
        headers={"xi-api-key": API_KEY, "Content-Type": "application/json"},
        json={
            "text": text,
            "model_id": "eleven_multilingual_v2",
            "voice_settings": {"stability": 0.5, "similarity_boost": 0.8}
        }
    )
    # احفظ الملف الصوتي مؤقتاً وارجع رابطه
    tmp = tempfile.NamedTemporaryFile(delete=False, suffix=".mp3", dir="static")
    tmp.write(response.content)
    tmp.close()
    base_url = os.environ.get("BASE_URL", "http://localhost:5000")
    return f"{base_url}/static/{os.path.basename(tmp.name)}"

def repeat_gather(msg):
    audio_url = text_to_speech(msg)
    resp = VoiceResponse()
    gather = Gather(input="speech", action="/process", language="ar-SA")
    gather.play(audio_url)
    resp.append(gather)
    return Response(str(resp), mimetype="text/xml")

if __name__ == "__main__":
    os.makedirs("static", exist_ok=True)
    app.run(host="0.0.0.0", port=5000, debug=False)

النشر على Railway (أسهل طريقة)

سيرفر دائم التشغيل بـ ~$5/شهر
1
ارفع الكود على GitHub

أنشئ Repo جديد على github.com وارفع الملفات: app.py و requirements.txt و .env

2
أنشئ مشروع على Railway

روح على railway.app → New Project → Deploy from GitHub → اختر الـ Repo

3
أضف متغيرات البيئة

Variables → أضف كل مفاتيح الـ API من ملف .env واحداً واحداً

4
احصل على رابط السيرفر

Settings → Generate Domain → ستحصل على رابط مثل https://smart-voice.railway.app

5
ربط Twilio بالسيرفر

روح لـ Twilio → Phone Numbers → رقمك → Voice Webhook: https://smart-voice.railway.app/call
